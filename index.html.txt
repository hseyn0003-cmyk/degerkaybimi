    <script>
        function hesapla() {
            const km = document.getElementById('km').value;
            const hasar = document.getElementById('hasar').value;
            const sonucAlani = document.getElementById('sonucAlani');
            const tutarP = document.getElementById('tutar');

            if(km && hasar) {
                let carpan = km > 150000 ? 0.30 : (km > 100000 ? 0.50 : 0.85);
                let sonuc = (hasar * carpan * 0.78).toLocaleString('tr-TR', { minimumFractionDigits: 2 });
                tutarP.innerText = sonuc + " ₺";
                sonucAlani.classList.remove('hidden');
            }
        }
