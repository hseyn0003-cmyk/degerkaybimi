        function hesapla() {
            const km = document.getElementById('km').value;
            const hasar = document.getElementById('hasar').value;
            const sonucDiv = document.getElementById('sonuc');

            if (km && hasar) {

                tutarP.innerText = new Intl.NumberFormat('tr-TR', { style: 'currency', currency: 'TRY' }).format(sonuc);
                sonucDiv.classList.remove('hidden');
            } else {
                alert("Lütfen tüm alanları doldurun.");
